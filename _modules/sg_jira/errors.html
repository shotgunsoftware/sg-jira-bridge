[docs]class InvalidSyncValue(ValueError):
    """
    Base class for exceptions raised when a value can't be translated to a valid
    value for a given field.
    """

    def __init__(self, field, value, *args, **kwargs):
        """
        :param str field: The Jira or PTR field for which the exception was raised.
        :param value: The Jira or PTR value for which the exception was raised.
        """
        super(InvalidSyncValue, self).__init__(*args, **kwargs)
        self._field = field
        self._value = value

    @property
    def field(self):
        """
        Return the field for which the exception was raised.
        """
        return self._field

    @property
    def value(self):
        """
        Return the value for which the exception was raised.
        """
        return self._value
